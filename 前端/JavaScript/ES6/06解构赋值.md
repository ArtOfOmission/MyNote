# 解构赋值
1. 左右两边结构一样
2. 右边必须是个东西
3. 声明和赋值不能分开
```javascript
let [a, b, c] = [1, 2, 3];
console.log(a + "/" + b + "/" + c);

let { d, e } = { d: 12, e: 23 };
console.log(e + "/" + e);

let [{ f, g }, [num1, num2, num3], num, str] = [{ f: 45, g: 34 }, [2, 3, 4], 22, 'sssss'];
console.log(f, g, num1, num2, num3, num, str);

let [json, arrN, num, str] = [{ f: 45, g: 34 }, [2, 3, 4], 22, 'sssss'];
console.log(json, arrN, num, str);
```
