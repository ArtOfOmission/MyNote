# 数组去重

## 双重for循环去重
```javascript
function unique(arr) {
    if (!Array.isArray(arr)) {
        console.log("type error!");
        return false;
    }
    let newArr = [arr[0]];
    for (let i = 0; i < arr.length; i++) {
        let isContain = false;
        for (let j = 0; j < newArr.length; j++) {
            if (arr[i] === newArr[j]) { //数组中已存在该元素
                isContain = true;
                break;
            }
        }
        if (!isContain) {
            newArr.push(arr[i]);
        }
    }
    return newArr;
}
```

## 单层for循环 + indexOf()方法

```javascript
function unique2(arr) {
    if (!Array.isArray(arr)) {
        console.log("type error!");
        return false;
    }
    let newArr = [];
    for (let i = 0; i < arr.length; i++) {
        if (newArr.indexOf(arr[i]) === -1) {
            newArr.push(arr[i]);
        }
    }
    return newArr;
}
```
