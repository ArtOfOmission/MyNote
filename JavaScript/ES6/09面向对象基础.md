# 继承

## 老式面向对象
```javascript
function User(name, pass) {
    this.name = name;
    this.pass = pass;
}
User.prototype.showNme = function () {
    console.log(this.name);
}
User.prototype.showPass = function () {
    console.log(this.pass);
}

function VipUser(name, pass, leval) {
    User.call(this, name, pass);
    this.leval = leval;
}
VipUser.prototype = new User();
VipUser.prototype.constructor = VipUser;
VipUser.prototype.showLevel = function () {
    console.log(this.leval);
}
let vip = new VipUser('cys', '123456', 3);
vip.showNme();
vip.showPass();
vip.showLevel();
```
## ES6面向对象
super--超类（父类）
 ```javascript
//ES6面向对象
class User {
    constructor(name, pass) {
        this.name = name;
        this.pass = pass;
    }
    showNme() {
        console.log(this.name);
    }
    showPass() {
        console.log(this.pass);
    }
}

class VipUser extends User {
    constructor(name, pass, level) {
        super(name, pass);//父类
        this.level = level;
    }
    showLevel() {
       console.log(this.level);
    }
}
var vip2 = new VipUser('sen', '123456', 2);
vip2.showNme();
vip2.showPass();
vip2.showLevel();
 ```
